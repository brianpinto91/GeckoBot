\begin{document}
\begin{tikzpicture}[
	part/.style={thick},
	anglearrow/.style={latex-latex, red!70},
	helpline/.style={gray},
	foot/.style={fill=gray},
	lettering/.style={red!80},
	distarrow/.style={|<->|, blue!50},
	distlettering/.style={blue!60},
	vec/.style={-latex, lightgray},
	vechead/.style={purple}
]

\def\rfoot{.1}
\def\rradi{.05}

\def\angledist{1}
\def\fontdist{.7}
\def\distdist{.2}
\def\fontdistdist{.5}


\def\alpi{35}
\def\alpii{40}
\def\gam{50}
\def\beti{60}
\def\betii{80}
\pgfmathsetmacro{\gamh}{\gam*.5}

\def\lleg{3}
\def\ltor{3.5}

\def\eps{30}
\pgfmathsetmacro{\ci}{\eps-\alpi-\gamh}
\pgfmathsetmacro{\cii}{\ci+\alpi+\beti}
\pgfmathsetmacro{\ciii}{180 + \gam + \ci + \alpi + \alpii}
\pgfmathsetmacro{\civ}{180 + \gam + \ci + \alpi - \betii}


\def\pi{3.14159}

\pgfmathsetmacro{\ri}{\lleg/\alpi*180/\pi}
\pgfmathsetmacro{\rii}{\lleg/\beti*180/\pi}
\pgfmathsetmacro{\rg}{\ltor/\gam*180/\pi}
\pgfmathsetmacro{\riii}{\lleg/\alpii*180/\pi}
\pgfmathsetmacro{\riv}{\lleg/\betii*180/\pi}


%% COS
\path (-8,-1) coordinate(O)node[below]{\{0\}};
\draw[-latex] (O)--++(1,0)node[below]{$x$};
\draw[-latex] (O)--++(0,1)node[left]{$y$};



\path (0,0) coordinate(F1);
%% F1
\draw[part] (F1)arc(180+\ci:180+\ci+\alpi:\ri)coordinate(OM);
\draw[helpline] (F1)--++(\ci:\ri)coordinate(R1)--(OM);
\draw[anglearrow] (R1)++(180+\ci:\angledist)arc(180+\ci:180+\ci+\alpi:\angledist);
\pgfmathsetmacro{\alpih}{\alpi*.5}
\path (R1)++(180+\ci+\alpih:\fontdist) node[lettering]{$\alpha_1$};
\draw[distarrow] (R1)++(180+\ci:\distdist+\ri)arc(180+\ci:180+\ci+\alpi:\distdist+\ri);
\path (R1)++(180+\ci+\alpih:\ri+\fontdistdist) node[distlettering]{$\ell_1$};


%% F2
\draw[part] (OM)arc(180+\ci+\alpi:180+\ci+\alpi+\beti:\rii)coordinate(F2);
\draw[helpline] (F2)--++(\cii:\rii)coordinate(R2)--(OM);
\draw[anglearrow] (R2)++(180+\cii:\angledist)arc(180+\cii:180+\cii-\beti:\angledist);
\pgfmathsetmacro{\betih}{\beti*.5}
\path (R2)++(180+\cii-\betih:\fontdist) node[lettering]{$\alpha_2$};
\draw[distarrow] (R2)++(180+\cii-\beti:\distdist+\rii)arc(180+\ci+\alpi:180+\ci+\alpi+\beti:\distdist+\rii);
\path (R2)++(180+\cii-\betih:\rii+\fontdistdist) node[distlettering]{$\ell_2$};


%% UM
\draw[part] (OM)arc(90+\ci+\alpi:90+\ci+\alpi+\gam:\rg)coordinate(UM);
\draw[helpline] (OM)--++(-90+\ci+\alpi:\rg)coordinate(Rg)--(UM);
\draw[anglearrow] (Rg)++(90+\ci+\alpi:\angledist)arc(90+\ci+\alpi:90+\ci+\alpi+\gam:\angledist);
\pgfmathsetmacro{\gamh}{\gam*.5}
\path (Rg)++(90+\ci+\alpi+\gamh:\fontdist) node[lettering]{$\alpha_3$};
\draw[distarrow] (Rg)++(90+\ci+\alpi:\distdist+\rg)arc(90+\ci+\alpi:90+\ci+\alpi+\gam:\distdist+\rg);
\path (Rg)++(90+\ci+\alpi+\gamh:\rg+\fontdistdist) node[distlettering]{$\ell_3$};
\path (Rg)++(90+\ci+\alpi+\gamh:\rg)coordinate(M);



%% F3
\draw[part] (UM)arc(\gam+\ci+\alpi:\gam+\ci+\alpi+\alpii:\riii)coordinate(F3);
\draw[helpline] (F3)--++(180+\gam+\ci+\alpi+\alpii:\riii)coordinate(R3)--(UM);
\draw[anglearrow] (R3)++(\gam+\ci+\alpi+\alpii:\angledist)arc(\gam+\ci+\alpi+\alpii:\gam+\ci+\alpi:\angledist);
\pgfmathsetmacro{\alpiih}{\alpii*.5}
\path (R3)++(\gam+\ci+\alpi+\alpiih:\fontdist) node[lettering]{$\alpha_4$};
\draw[distarrow] (R3)++(\gam+\ci+\alpi:\distdist+\riii)arc(\gam+\ci+\alpi:\gam+\ci+\alpi+\alpii:\distdist+\riii);
\path (R3)++(\gam+\ci+\alpi+\alpiih:\riii+\fontdistdist) node[distlettering]{$\ell_4$};



%% F4
\draw[part] (UM)arc(\gam+\ci+\alpi:\gam+\ci+\alpi-\betii:\riv)coordinate(F4);
\draw[helpline] (F4)--++(180+\gam+\ci+\alpi-\betii:\riv)coordinate(R4)--(UM);
\draw[anglearrow] (R4)++(\gam+\ci+\alpi-\betii:\angledist)arc(\gam+\ci+\alpi-\betii:\gam+\ci+\alpi:\angledist);
\pgfmathsetmacro{\betiih}{\betii*.5}
\path (R4)++(\gam+\ci+\alpi-\betiih:\fontdist) node[lettering]{$\alpha_5$};
\draw[distarrow] (R4)++(\gam+\ci+\alpi:\distdist+\riv)arc(\gam+\ci+\alpi:\gam+\ci+\alpi-\betii:\distdist+\riv);
\path (R4)++(\gam+\ci+\alpi-\betiih:\riv+\fontdistdist) node[distlettering]{$\ell_5$};



%% Constant
\draw[helpline] (F1)--++(0:\angledist+.2);
\draw[anglearrow] (F1)++(0:\angledist)arc(0:360+\ci:\angledist);
\pgfmathsetmacro{\cih}{\ci*.5}
\path (F1)++(180+\cih:\fontdist) node[lettering]{$\varphi_1$};

\draw[helpline] (F2)--++(0:\angledist+.2);
\draw[anglearrow] (F2)++(0:\angledist)arc(0:\cii:\angledist);
\pgfmathsetmacro{\ciih}{\cii*.5}
\path (F2)++(\ciih:\fontdist) node[lettering]{$\varphi_2$};

\draw[helpline] (F3)--++(0:\angledist+.2);
\draw[anglearrow] (F3)++(0:\angledist)arc(0:\ciii:\angledist);
\pgfmathsetmacro{\ciiih}{\ciii*.5}
\path (F3)++(\ciiih:\fontdist) node[lettering]{$\varphi_3$};


\draw[helpline] (F4)--++(0:\angledist+.2);
\draw[anglearrow] (F4)++(0:\angledist)arc(0:\civ:\angledist);
\pgfmathsetmacro{\civh}{\civ*.5}
\path (F4)++(\civh:\fontdist) node[lettering]{$\varphi_4$};


\draw[helpline] (M)--++(0:\angledist+.2);
\draw[helpline] (M)--++(\eps:\angledist+.2);
\draw[anglearrow] (M)++(0:\angledist)arc(0:\eps:\angledist);
\pgfmathsetmacro{\epsh}{\eps*.5}
\path (M)++(\epsh:\fontdist) node[lettering]{$\varepsilon$};




\draw[foot] (F1)circle(\rfoot)++(\ci+90:\fontdistdist)node{$F1$};
\draw[foot] (F2)circle(\rfoot)++(\cii-90:\fontdistdist)node{$F2$};
\draw[foot] (F3)circle(\rfoot)++(\ciii-90:\fontdistdist)node{$F3$};
\draw[foot] (F4)circle(\rfoot)++(\civ+90:\fontdistdist)node{$F4$};

\draw[foot] (OM)circle(\rfoot)++(\ci+\alpi:\fontdistdist)node{$OM$};
\draw[foot] (UM)circle(\rfoot)++(\ciii-\alpii:\fontdistdist)node{$UM$};

\draw[foot] (M)circle(\rfoot)++(90+\ci+\alpi+\gamh:-\fontdistdist)node{$M$};

\draw (R1)circle(\rradi)++(\ci+\alpih:\fontdistdist)node{$R1$};
\draw (R2)circle(\rradi)++(\cii-\betih:\fontdistdist)node{$R2$};
\draw (Rg)circle(\rradi)++(\ci+\alpi+\gamh-90:\fontdistdist)node{$RG$};
\draw (R3)circle(\rradi)++(\ciii-\alpiih:\fontdistdist)node{$R3$};
\draw (R4)circle(\rradi)++(\civ+\betiih:\fontdistdist)node{$R4$};



%% Vectors
\draw[vec](O)--(F1)node[vechead, near start, above]{$\bm{p}_1$};
\draw[vec](O)--(F2)node[vechead, pos=.3, above]{$\bm{p}_2$};
\draw[vec](O)--(F3)node[vechead, pos=.5, left]{$\bm{p}_3$};
\draw[vec](O)--(F4)node[vechead, pos=.2, right]{$\bm{p}_4$};



\end{tikzpicture}
